Claude Web Scraper
1,500 Salesforce accounts with unknown ERPs. AEs burning 10 minutes per account doing manual ERP research. Clay was out of credits. I used Claude to build a Python web scraper that checked company websites for portal links, job listings and page source for ERP signals. Built it out over a few afternoons. Ran it overnight. Delivered a confirmed target list to RevOps by morning.
import requests, csv, time
from bs4 import BeautifulSoup

PORTAL_HINTS = ["portal.netsuite",
                "wdaisy", "intacct.com"]
CAREER_TERMS  = ["NetSuite Admin",
                 "Workday Financials"]

def detect_erp(domain):
    url = f"https://{domain}"
    try:
        r    = requests.get(url, timeout=8)
        soup = BeautifulSoup(r.text, "html.parser")
        links = [a.get("href","")
                 for a in soup.find_all("a")]
        for link in links:
            for hint in PORTAL_HINTS:
                if hint in link:
                    return hint
        jobs = requests.get(
            f"{url}/careers", timeout=8)
        for term in CAREER_TERMS:
            if term in jobs.text:
                return term
    except: pass
    return "unknown"

with open("accounts.csv") as f:
    accounts = list(csv.DictReader(f))

results = []
for acct in accounts:
    erp = detect_erp(acct["domain"])
    results.append({**acct, "erp": erp})
    time.sleep(0.4)

# 1,500 accounts. Ran overnight.
print(f"Done. {len(results)} processed.")
